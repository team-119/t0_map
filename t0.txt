function getLocation()
{
    // 브라우저 Geolocation API 지원여부 확인
    if (navigator.geolocation) { // 지원
        navigator.geolocation.getCurrentPosition(getPosition);
        
   } else { // 지원안함
        alert("Oops! Browser can't load location.");
    }
}

function getPosition(position)
{
  var myLat = position.coords.latitude //현재위도
  var myLon = position.coords.longitude //현재경도
    document.getElementById("location").innerHTML = 
        "Latitude: " + myLat + "<br>" + "Longitude: " + myLon
